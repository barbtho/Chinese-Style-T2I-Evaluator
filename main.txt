import streamlit as st
from evaluate2 import TextToImageEvaluator
import time


def main():
    # 设置页面标题和布局
    st.set_page_config(
        page_title="文生图模型评测系统",
        page_icon="",
        layout="wide"
    )

    # 标题
    st.markdown(
        "<h1 style='text-align: center; color: #1f77b4;'>文生图模型评测系统</h1>",
        unsafe_allow_html=True
    )

    # 初始化评估器
    if 'evaluator' not in st.session_state:
        with st.spinner("正在初始化模型，这可能需要几分钟..."):
            st.session_state.evaluator = TextToImageEvaluator()

    # 输入板块
    st.header("输入设置")

    input_method = st.radio(
        "选择输入方式:",
        ["单条文本输入", "批量文件输入"],
        horizontal=True
    )

    prompts = []

    if input_method == "单条文本输入":
        prompt = st.text_area(
            "请输入文本描述:",
            placeholder="输入您想要生成的图像描述（支持中文、英文）",
            height=100
        )
        if prompt:
            prompts = [prompt]

    else:  # 批量文件输入
        uploaded_file = st.file_uploader(
            "选择文本文件（每行一个提示词）",
            type=['txt'],
            help="上传一个文本文件，每行包含一个提示词"
        )

        if uploaded_file is not None:
            # 读取文件内容
            stringio = uploaded_file.getvalue().decode("utf-8")
            prompts = [line.strip() for line in stringio.splitlines() if line.strip()]
            st.info(f"已读取 {len(prompts)} 个提示词")

    # 参数设置板块
    st.header("参数设置")

    col1, col2, col3 = st.columns(3)

    with col1:
        width = st.selectbox(
            "图像宽度:",
            [256, 512, 768, 1024],
            index=1
        )

    with col2:
        height = st.selectbox(
            "图像高度:",
            [256, 512, 768, 1024],
            index=1
        )

    with col3:
        num_inference_steps = st.slider(
            "推理步数:",
            min_value=5,
            max_value=50,
            value=10,
            help="步数越多，图像质量可能越高，但生成时间越长"
        )

    guidance_scale = st.slider(
        "引导尺度 (guidance_scale):",
        min_value=1.0,
        max_value=20.0,
        value=7.5,
        step=0.5,
        help="控制生成图像与文本的匹配程度，值越大匹配度越高"
    )

    # 开始评估按钮
    if st.button("开始评测", type="primary", disabled=len(prompts) == 0):
        if len(prompts) == 0:
            st.error("请先输入提示词")
            return

        # 处理状态显示
        st.header("处理状态")

        progress_bar = st.progress(0)
        status_text = st.empty()
        log_container = st.container()

        all_results = []

        for i, prompt in enumerate(prompts):
            # 更新进度和状态
            progress = (i / len(prompts)) * 100
            progress_bar.progress(int(progress))
            status_text.text(f"处理进度: {i + 1}/{len(prompts)}")

            with log_container:
                st.subheader(f"处理: '{prompt}'")

                # 步骤1: 等待输入（已完成）
                st.info("完成输入")

                # 步骤2: 生成图像
                with st.spinner("正在生成图像..."):
                    image, english_prompt = st.session_state.evaluator.generate_image_from_text(
                        prompt, width, height, num_inference_steps, guidance_scale
                    )
                st.success("完成图像生成")

                # 步骤3: 图生文
                with st.spinner("正在生成图像描述..."):
                    generated_text = st.session_state.evaluator.generate_text_from_image(image)
                st.success("完成图像描述")

                # 步骤4: 计算分数
                with st.spinner("正在计算评测分数..."):
                    semantic_similarity = st.session_state.evaluator.calculate_semantic_similarity(
                        english_prompt, generated_text
                    )
                    clip_score = st.session_state.evaluator.calculate_clip_score(image, english_prompt)
                st.success("完成分数计算")


                # 保存结果
                result = {
                    "original_text": prompt,
                    "english_text": english_prompt,
                    "generated_image": image,
                    "generated_text": generated_text,
                    "semantic_similarity": semantic_similarity,
                    "clip_score": clip_score
                }
                all_results.append(result)

        # 更新最终进度
        progress_bar.progress(100)
        status_text.text("完成所有处理")

        # 存储结果到session state
        st.session_state.results = all_results

        # 自动跳转到结果展示
        st.success(f"成功处理 {len(all_results)} 个提示词")

    # 结果展示
    if 'results' in st.session_state and st.session_state.results:
        st.header("评测结果")

        for i, result in enumerate(st.session_state.results):
            st.subheader(f"结果 {i + 1}")

            col1, col2 = st.columns([1, 1])

            with col1:
                st.image(result['generated_image'], caption="生成的图像", use_container_width=True)

            with col2:
                # 创建结果表格
                st.markdown("### 评估详情")

                # 原始文本
                st.text_area(
                    "原始文本:",
                    value=result['original_text'],
                    height=60,
                    key=f"original_{i}"
                )

                # 翻译文本
                translation_value = result['english_text'] if result['english_text'] != result[
                    'original_text'] else "\\"
                st.text_area(
                    "翻译文本:",
                    value=translation_value,
                    height=60,
                    key=f"translation_{i}"
                )

                # 生成描述
                st.text_area(
                    "生成描述:",
                    value=result['generated_text'],
                    height=60,
                    key=f"generated_{i}"
                )

                # 语义相似度
                st.metric(
                    "语义相似度",
                    f"{result['semantic_similarity']:.4f}",
                    delta=None,
                    delta_color="normal"
                )

                # CLIP分数
                st.metric(
                    "CLIP分数",
                    f"{result['clip_score']:.4f}",
                    delta=None,
                    delta_color="normal"
                )

            st.divider()


if __name__ == "__main__":
    main()
